Permit.io is a cloud-based authorization service that allows you to define and manage permissions for your application.
In order to make it easier and safer to manage your objects and policies in Permit.io, we have created a Terraform provider.

## Usage

### Provider Definition

```hcl
terraform {
  required_providers {
    permitio = {
      source  = "permitio/permitio"
      version = "~> 0.0.1"
    }
  }
}
```


### Configure the AWS Provider
```hcl
provider "aws" {
    api_url = "https://api.permit.io"
    api_key = "YOUR_API_KEY"
}
```

### Creating Objects in Permitio

#### Create a Resource
```hcl
resource "permit_resource" "document" {
    key = "document"
    name = "Document"
    description = "A confidential document"
    actions = {
        "read": {
            "name": "Read",
            "description": "Read a document",
            "attributes": ["title"]
        },
        "write": {
            "name": "Write",
            "description": "Write a document",
            "attributes": ["title"]
        }
    attributes = [{
        "title":{
            "type": "string",
            "description": "The title of the document"
        }
    }]
}
```

#### Create a Role

```hcl
resource "permit_role" "reader" {
    key = "reader"
    name = "Reader"
    description = "A role that allows reading documents"
    permissions = [
        "document:read"
    ]
    attributes = {
        "title": {
            "type": "string",
            "description": "The title of the document"
    }
}
```
