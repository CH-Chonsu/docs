## Usage

### Provider Definition

```hcl
terraform {
  required_providers {
    permitio = {
      source  = "permitio/permitio"
      version = "~> 0.0.1"
    }
  }
}
```


### Configure the AWS Provider
```hcl
provider "aws" {
    api_url = "https://api.permit.io"
    api_key = "YOUR_API_KEY"
}
```

### Creating Objects in Permitio

#### Create a Resource
```hcl
# Create a Resource
resource "permit_resource" "document" {
    key = "document"
    name = "Document"
    description = "A confidential document"
    actions = [{
        name = "Read"
        description = "Read the document"
        attributes = [{
            "created":{
                "type": "date",
                "description": "The date the document was created"
            }
        }]
    }]
    attributes = [{
        "title":{
            "type": "string",
            "description": "The title of the document"
        }
    }]
}
```
